解構元與拷貝建構元

這個程式先定義了一個名為 CWin 的視窗類別，並在裡面宣告了一些變數。然後，定義一個拷貝的建構元，用來初始化物件。接下來，定義一個解構元，用來釋放物件所佔用的記憶體空間。最後顯示物件的屬性。讓我學習到如何在 C++ 中建立和使用類別，以及如何管理記憶體。



程式碼

#include
#include
using namespace std;
class CWin //定義視窗類別
{
private:
char id,*title; //宣告title為指向字元陣列的指標
int width, height;
public:
CWin(char i, int w, int h,char* text=(char *)"Default window") :id(i), width(w), height(h)
{
title = new char[strlen(text) + 1];
strcpy(title, text); //將text指向的字串拷貝到title陣列裡
cout << "建構元被呼叫了....." << endl;
}
CWin(const CWin& win) //定義拷貝建構元
{
cout << "拷貝建構員呼叫了...." << endl;
id = win.id; //拷貝資料成員
width = win.width;
height = win.height;
title = new char[strlen(win.title) + 1]; //配置記憶空間
strcpy(title, win.title);
}
~CWin() //解構元的原型
{
cout << "解構原被呼叫了 ，Win" << this->id << "被銷毀了..." << endl;
delete[] title; //釋放title所指向的記憶空間
system("pause");
}
void show_member(void) //顯示id與title成員
{
cout << "Windows " << id << ":" << title << endl;
cout << "Width=" << width << ",Height=" << height << endl;
}
};
int main()
{
CWin win1('A', 50, 40,(char *)"Main Window"); //建立win1物件
CWin win2('B', 40, 50);
CWin win3('C', 60, 70);
CWin win4('D', 90, 40);
CWin win5(win1); //呼叫拷貝建構元
cout << "sizeof(win1)=" << sizeof(win1) << endl;
cout << "sizeof(win2)=" << sizeof(win2) << endl;
win1.show_member();
win2.show_member();
win5.show_member();
system("pause");
return 0;
}

基本陣列操作與數據處理

計算陣列大小，找出陣列中的最大和最小值，讓使用者輸入分數並計算平均值。程式還處理和顯示了二維和三維陣列的數據，並使用函數來顯示學生成績和計算平均成績。這讓我了解如何使用C++來操作和處理各種數據。



程式碼

#include
#include
using namespace std;
// 定義常數
#define MAX 5
#define SIZE 5
// 函數宣告
void show(int[]);
double average(int[]);
int main()
{
// 定義長整型陣列
long long score[6];
cout << "sizeof(score[1]=" << sizeof(score[1]) << endl; // 顯示陣列元素的大小
cout << "sizeof(score)" << sizeof(score) << endl; // 顯示整個陣列的大小
// 定義並初始化整型陣列
int A[] = { 48,70,30,17,62 };
int i, min = A[0], max = A[0];
int length = sizeof(A) / sizeof(int); // 計算陣列長度
cout << "Elements in array A are";
// 迴圈遍歷陣列元素
for (i = 0; i < length; i++)
{
cout << A[i] << " "; // 顯示元素
if (A[i] > max) // 找最大值
max = A[i];
if (A[i] < min) // 找最小值
min = A[i];
}
// 顯示最大值和最小值
cout << endl << "Maximum is " << max;
cout << endl << "Minimum is " << min << endl;
// 定義一個新的整型陣列
int score1[MAX];
int ii = 0, num;
float sum = 0.0f;
cout << "Enter -1 stopping input" << endl;
// 迴圈輸入成績，直到輸入-1或達到最大空間
do
{
if (ii == MAX)
{
cout << "No more space!!" << endl;
ii++;
break;
}
cout << "Input score:";
cin >> score1[ii];
} while (score1[ii++] >= 0);
num = ii - 1; // 計算有效的輸入數
for (ii = 0; ii < num; ii++) // 計算總和
sum += score1[ii];
cout << "Average of all is" << sum / num << endl; // 顯示平均值
// 定義2x4的整型陣列並初始化
int i1, j1, sum1 = 0;
int sale[2][4] = { {30,35,26,32},{33,34,30,29} };
// 迴圈遍歷陣列並計算總和
for (i1 = 0; i1 < 2; i1++)
{
cout << "業務員" << (i1 + 1) << "的業績分別為 ";
for (j1 = 0; j1 < 4; j1++)
{
cout << sale[i1][j1] << " ";
sum1 += sale[i1][j1];
}
cout << endl;
}
cout << "本年度總銷售量為:" << sum1 << "輛車" << endl;
// 定義3D整型陣列並初始化 int AA[2][4][3] = { {{21,32,65},{78,94,76},{79,44,65},{89,54,73}},{{32,56,89},{43,23,32},{32,56,78},{94,78,45}} };
int i2, j2, k2, max2 = AA[0][0][0];
// 迴圈遍歷3D陣列並找出最大值
for (i2 = 0; i2 < 2; i2++)
for (j2 = 0; j2 < 4; j2++)
for (k2 = 0; k2 < 3; k2++)
if (max2 < AA[i2][j2][k2])
max2 = AA[i2][j2][k2];
cout << "max=" << max2 << endl; // 顯示最大值
// 定義並初始化整型陣列
int score2[SIZE] = { 89,54,73,95,71 };
cout << "學生成績為 ";
show(score2); // 顯示成績
cout << "平均成績=" << average(score2) << endl; // 顯示平均成績
system("pause");
}
// 顯示陣列元素的函數
void show(int a[])
{
for (int i = 0; i < SIZE; i++)
cout << a[i] << " ";
cout << endl;
return;
}
// 計算陣列平均值的函數
double average(int a[])
{
double sum = 0;
for (int i = 0; i < SIZE; i++)
sum += a[i];
return (sum / SIZE);
}
